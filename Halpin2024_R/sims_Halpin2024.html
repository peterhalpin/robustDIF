# Laod sim data
load("sim.200.01.19.2025.RData")

sim.list <- c(sim.list.200, sim.list.350, sim.list.500)

# Parms
n.reps <- 500
levels <- 0:10
n.levels <- 11
sample.sizes <- c(200, 350, 500)
n.sample.sizes <- 3

# Format RDIF results
sim.rdif <- Reduce(rbind, lapply(sim.list, function (x) x$rdif))
sim.rdif$n.persons <- rep(sample.sizes, each = n.reps*n.levels)
sim.rdif$n.dif <- rep(rep(levels, each = n.reps), times = n.sample.sizes)

# Format LRT results
sim.lrt <- lapply(sim.list, function (x) x$lrt)
sim.lrt <- matrix(unlist(sim.lrt), 
                  ncol = 8, 
                  nrow = n.reps*n.levels*n.sample.sizes, 
                  byrow = T) |> 
            data.frame()
sim.lrt$n.persons <- rep(sample.sizes, each = n.reps*n.levels)
sim.lrt$n.dif <- rep(rep(levels, each = n.reps), times = n.sample.sizes)
names(sim.lrt) <- names(sim.rdif)

# Format True DIF pattern results
sim.true <- lapply(sim.list, function (x) x$true.delta.hat)
sim.true <- matrix(unlist(sim.true), 
                   ncol = 8, 
                   nrow = n.reps*n.levels*n.sample.sizes, 
                   byrow = T) |> 
            data.frame()
sim.true$n.persons <- rep(sample.sizes, each = n.reps*n.levels)
sim.true$n.dif <- rep(rep(levels, each = n.reps), times = n.sample.sizes)
names(sim.true) <- names(sim.rdif)

# Stack
sim <- rbind(sim.rdif, sim.lrt, sim.true)
sim$method <- rep(c("RDIF", "LRT", "True"), each = n.reps*n.levels*n.sample.sizes)
sim.scale <- lapply(sim.list, function (x) x$true.scale) |> unlist()
sim$true.scale <- rep(sim.scale, times = 3)

Bias and precision

Figure 2 from the paper

temp.bias <- sim |> 
  group_by(method, n.persons, n.dif) |> 
  reframe(bias.naive = naive.est - true.scale,
          bias.dif = rdif.est - true.scale) |> 
  data.frame()

# Naive not run when no items exhibit DIF on LRT/True, replicate from RDIF
naive <- cbind(rep("Naive", times = n.sample.sizes*n.levels),  
                     temp.bias[temp.bias$method == "RDIF", c(2:4)])

naive <- rbind(naive, naive, naive)
other <- temp.bias[-4] 
names(naive) <- names(other) <- c("Method", "n.persons", "N.DIF", "Bias") 

# Set DIF bias to Naive bias when the former is missing (due to not run) 
index <- is.na(other$Bias)
other$Bias[index] <- naive$Bias[index]

sim.bias <- rbind(other, naive)
sim.bias$Method <- 
  factor(
    sim.bias$Method, 
    levels = c("Naive", "LRT", "RDIF", "True"),
    ordered = T)

Figure 3 from the paper.

sim.power <- sim |> 
  group_by(method, n.persons, n.dif) |> 
  summarize(p.val = sum(p.val < .05, na.rm = T)/n.reps) |> 
  data.frame()
